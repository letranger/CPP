#include <stdio.h>
#include <string.h>
#include <stdlib.h>
int main(int argc, const char * argv[]) {
    // insert code here...
    char  poker[53][5];
    char m[4] = {'A','B','C','D'};
    char tmp[5];
    int i, j, k;
    for (i = 0; i < 4; i++) {
        for (j = 0; j < 13; j++) {
            k = 13 * i + j;
            sprintf(tmp, "%d%c", j+1, m[i]);
            //printf("%d: %s\n", k, tmp);
            strcpy(poker[k], tmp);
        }
    }
    for (i = 0; i < 52; i++) {
        printf("%s\n", poker[i]);
    }

    return 0;
}
